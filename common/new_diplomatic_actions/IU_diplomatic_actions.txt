IU_add_to_sphere_of_influence = {
	category = influence
	
	alert_index = 41
	alert_tooltip = IU_sphere_tooltip
	
	require_acceptance = no
	
	is_visible = {
		is_subject = no 
		FROM = {
			is_subject = no
		}
		OR = {
			has_idea_group = exploration_ideas
			AND = {
				tag = TUR 
				any_neighbor_province = {
					OR = {
						region = arabian_sea_region
						region = western_indian_ocean_region
					}
				}
			}
		}
	}
	is_allowed = {
		FROM = {
			OR = {
				technology_group = indian
				technology_group = chinese
			}
			NOT = { total_development = ROOT }
		}
		
		check_variable = {
			which = IU_indian_influence
			value = 0
		}
	}
	
	on_accept = {
		create_subject = {
			subject_type = IU_sphere_subject
			subject = FROM 
		}
	}
}
