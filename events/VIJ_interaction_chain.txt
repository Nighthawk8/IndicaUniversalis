namespace = vijinteraction

country_event = {
	id = vijinteraction.1
	title = "vijinteraction.NAME1"
	desc = "vijinteraction.DESC1"
	picture = BANKRUPTCY_eventPicture
	
	trigger = {
		OR = {
			tag = MAB
			tag = VND
			tag = KOC
			tag = KLT
		}
		VIJ = {
			exists = yes
		}
		if = {
			limit = {
				VIJ  = {
					exists = yes
				}
			}
		variable_arithmetic_trigger = {
			custom_tooltip = VIJ_HAS_LESS_PROVINCES
			
			export_to_variable = {
				variable_name = my_provinces
				value = trigger_value:num_of_cities
			}
			export_to_variable = {
				variable_name = vij_provinces
				value = trigger_value:num_of_cities
				who = VIJ
			}
			
			check_variable = {
				which = vij_provinces
				which = my_provinces
			}
		}
		variable_arithmetic_trigger = {
			custom_tooltip = VIJ_HAS_LESS_DEV
			
			export_to_variable = {
				variable_name = my_dev
				value = trigger_value:total_development
			}
			export_to_variable = {
				variable_name = vij_dev
				value = trigger_value:total_development
				who = VIJ
			}
			
			check_variable = {
				which = vij_dev
				which = my_dev
			}
		}
		}
		NOT = {
			OR = {
			has_country_modifier = has_truce_with_vij
			has_country_modifier = rejects_truce_with_vij
			}
			ROOT = {
				war_with = VIJ
			}
			truce_with = VIJ
		}
	}
	
	mean_time_to_happen = {
		days = 1		
	}

	option = {	# stand alone
		name = "vijinteraction.1a"
		ai_chance = { factor = 25 }
		add_mil_power = 100
		hidden_effect = {
			add_country_modifier = {
				name = rejects_truce_with_vij
				duration = 3650
			}
		}
	}

	option = {	# gains truce
		name = "vijinteraction.1b"
		ai_chance = { factor = 75 }
		add_dip_power = -50
		add_truce_with = VIJ
		hidden_effect = {
			add_country_modifier = {
				name = has_truce_with_vij
				duration = 1825
			}
		}
	}
}
