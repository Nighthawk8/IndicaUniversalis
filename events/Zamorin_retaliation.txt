province_event = {
	id = zamorinretaliation.1
	title = "zamorinretaliation.NAME1"
	desc = "zamorinretaliation.DESC1"
	picture = LAND_MILITARY_eventPicture
	
	trigger = {
		has_province_modifier = lvl_3_influence_province
		owner = {
			NOT = {
				tag = MAB
			}
		}
	}
	
	mean_time_to_happen = {
		months = 60
	}

	option = {	# remove
		name = "zamorinretaliation.1a"
		ai_chance = { factor = 50 }
		remove_province_modifier = lvl_3_influence_province
		add_province_modifier = {
			name = lvl_2_influence_province
			duration = -1
		}
		add_adm_power = -50
		if = {
			limit= {
				province_id = 4413
			}
		country_event = {
			id = zamorinretaliation.100
		}
		hidden_effect = {
		subtract_variable = {
			which = 4413influencelvl
			value = 1
			}
		}
		}
		if = {
			limit= {
				province_id = 533
			}
		country_event = {
			id = zamorinretaliation.101
		}
		hidden_effect = {
		subtract_variable = {
			which = 533influencelvl
			value = 1
			}
		}
		}
		if = {
			limit= {
				province_id = 538
			}
		country_event = {
			id = zamorinretaliation.102
		}
		hidden_effect = {
		subtract_variable = {
			which = 538influencelvl
			value = 1
			}
		}
		}
		if = {
			limit= {
				province_id = 535
			}
		country_event = {
			id = zamorinretaliation.103
		}
		hidden_effect = {
		subtract_variable = {
			which = 535influencelvl
			value = 1
			}
		}
		}
	}

	option = {	# leave
		name = "zamorinretaliation.1b"
		ai_chance = { factor = 50 }
		add_dip_power = 50
	}
}
